int binary_s(char arr[], char search, int l, int h)
{
    int mid = (l + h) / 2;
    if (arr[mid] == search)
    {
        return 1;
    }
    else if (mid == l && mid == h)
    {
        return -1;
    }
    else if (search > arr[mid])
    {
        binary_s(arr, search, mid + 1, h);
    }
    else
    {
        binary_s(arr, search, l, mid - 1);
    }
}
