int binary_s(char arr[], char search)
{
    int l = 0, h = 7, mid;
    while (l <= h)
    {
        mid = l + h / 2;
        if (search == arr[mid])
        {
            return 1;
        }
        else
        {
            if (search < arr[mid])
            {
                h = mid - 1;
            }
            else if (search > arr[mid])
            {
                l = mid + 1;
            }
        }
    }
}
