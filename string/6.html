//  Enter the string from user and find the length of the string
#include 

long strlen(char str[])
{
    int len;
    for (len = 0; str[len] != '\0'; len++)
        ;
    return len;
}
int main()
{
    char str[40];
    printf("Enter a String\n");
    gets(str);
    printf("Length of the string : %ld\n", strlen(str));
    return 0;
}
