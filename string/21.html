// print all the permutation of a string i.e., "ABCD"
#include 
void possiblePair(char str[], int n)
{
    int i, j, k;
    for (i = 0; i < n; i++)
    {
        for (j = i + 1; j < n; j++)
        {
            for (k = i; k <= j; k++)
            {
                printf("%c", str[k]);
            }
            printf("\n");
        }
    }
}
int main()
{
    char str[] = "abcd";
    int n = sizeof(n);
    possiblePair(str, n);

    return 0;
}

Output:

ab  
abc 
abcd
bc  
bcd 
cd
