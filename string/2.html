// Enter the string from user and convert it in lower case
#include 
char *Lcase(char *str)
{
    int i;
    for (i = 0; str[i] != '\0'; i++)
    {
        if (str[i] >= 'A' && str[i] <= 'Z')
        {
            str[i] = str[i] + 32;
        }
    }
    return str;
}
int main()
{
    char str[50];
    printf("Enter the String :- ");
    gets(str);
    printf("The uppercase string is %s\n", Lcase(str));
}

Enter the String :- HELLOO
