// Enter a string form user and copy it into another string variable
#include 

char *strCopy(char dest[], char src[])
{
    int i;
    for (i = 0; src[i] != '\0'; i++)
    {
        dest[i] = src[i];
    }
    dest[i] = src[i];
    return dest;
}

{
