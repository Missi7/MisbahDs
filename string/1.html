// Enter the string from user and convert it in upper case
#include 
char *Ucase(char *str)
{
    int i;
    for (i = 0; str[i] != '\0'; i++)
    {
        if (str[i] >= 'a' && str[i] <= 'z')
        {
            str[i] = str[i] - 32;
        }
    }
    return str;
}
int main()
{
    char str[50];
    printf("Enter the String :- ");
    gets(str);
    printf("The uppercase string is %s\n", Ucase(str));
}

Enter the String :- Hello World
The uppercase string is HELLO WORLD
