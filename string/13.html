// Enter string from user and copy reverse of the string into another string

#include 

char *strrevcpy(char dest[], char src[])
{
    int i = 0, j = 0;
    while (src[j] != '\0')
        j++;

    while (src[i] != '\0')
    {
        dest[i++] = src[--j];
    }
    dest[i] = src[i];
    return dest;
}

int main()
{
    char str[30], copiedstr[30];
    printf("Enter String\n");
    gets(str);

    printf("Copied String is %s\n", strrevcpy(copiedstr, str));
    return 0;
}
