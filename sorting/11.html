#define final_length 100
#define MAX_STRINGS 10

// Function to perform Selection Sort on a string array
void selection(char ary[][final_length], int n)
{
    for (int i = 0; i < n - 1; i++)
    {
        int min_index = i;
        for (int j = i + 1; j < n; j++)
        {
            // it returns -ve value if first string is smaller than other
            if (strcmp(ary[j], ary[min_index]) < 0)
            {
                min_index = j;
            }
        }
        // Swap the minimum element with the current element
        char temp[final_length];
        strcpy(temp, ary[i]);
        strcpy(ary[i], ary[min_index]);
        strcpy(ary[min_index], temp);
    }
}

int main()
{
    int n;
    printf("Enter the number of strings in the array: ");
    scanf("%d", &n);

    if (n > MAX_STRINGS)
    {
        printf("Exceeded maximum allowed strings.\n");
        return 1;
    }

    char input[MAX_STRINGS][final_length];
    printf("Enter %d strings:\n", n);
    for (int i = 0; i < n; i++)
    {
        scanf("%s", input[i]);
    }

    // Record the start time
    clock_t start = clock();

    // Perform Selection Sort
    selection(input, n);

    // Record the end time
    clock_t end = clock();

    // Calculate the time taken in seconds
    double time_complexcity = ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Sorted strings using Selection Sort:\n");
    for (int i = 0; i < n; i++)
    {
        printf("%s\n", input[i]);
    }
    printf("Time taken: %f seconds\n", time_complexcity);

    return 0;
}
