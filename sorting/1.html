// ENTER A INTEGER ARRAY FROM THE USER, ENTER AN INTEGER VALUE AND PERFORM LINEAR SEARCH ALGORITHM THROUGH ITERATION.

#include 
void Lsearch(int arr[], int size, int i)
{
    int key;
    printf("Enter the value you want to search:-");
    scanf("%d", &key);
    for (i = 0; i < size; i++)
    {
        if (arr[i] == key)
        {
            printf("The value is at %d position in array", i + 1);
            break;
        }
    }
    if (i == 5)
    {
        printf("The value you entered not found!!!!!!");
    }
}

int main()
{
    int arr[5], size = 5, i, key;

    for (i = 0; i < size; i++)
    {
        printf("Enter the Value:- ");
        scanf("%d", &arr[i]);
    }
    Lsearch(arr, size, i);
    return 0;
}
