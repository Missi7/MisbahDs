// Enter number from user and add all digit of number using recursion
#include 

int sumOfDigits(int num)
{
    if (num == 0)
    {
        return 0;
    }
    else
    {
        return num % 10 + sumOfDigits(num / 10);
    }
}

int main()
{
    int number;

    printf("Enter a number: ");
    scanf("%d", &number);

    int sum = sumOfDigits(number);

    printf("The sum of the digits of %d is %d\n", number, sum);

    return 0;
}


    Output:


Enter a number: 23
The sum of the digits of 23 is 5
