// Enter an array from user and find the k-th smallest value..
#include 
void main()
{
    int arr[5], n = 5, i, j, temp, k;
    printf("Enter the values in array\n");
    for (i = 0; i < n; i++)
    {
        printf("Enter the value: ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < n; i++)
    {
        for (j = i + 1; j < n; j++)
        {
            if (arr[i] > arr[j])
            {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
    printf("The sorted array :-\n");
    for (i = 0; i < n; i++)
    {
        printf("%d\n", arr[i]);
    }
    printf("Enter the k-th value\n");
    scanf("%d", &k);
    if (k >= 1 && k <= n)
    {
        printf("k-th smallest value of given list is %d", arr[k - 1]);
    }
    else
    {
        printf("we have only 5 elements in array...");
    }
}
