// Find the local maxima in an array.
#include 
void main()
{
    int arr[5], n = 5, i;
    printf("Enter the array :-\n");
    for (i = 0; i < n; i++)
    {
        printf("Enter the value:");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < n; i++)
    {
        if (arr[i] > arr[i - 1] && arr[i] > arr[i + 1])
        {
            printf("%d is the local maxima\n", arr[i]);
        }
    }
}
