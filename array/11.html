// Enter an array from user and find the duplicate values
#include 
void main()
{
    int arr[5], n = 5, i, j, avail=0;
    for (i = 0; i < n; i++)
    {
        printf("Enter the value : ");
        scanf("%d", &arr[i]);
    }
    printf("Duplicate values\n");
    for (i = 0; i < n; i++)
    {
        for (j = i+ 1; j < n; j++)
        {
            if (arr[i] == arr[j])
            {
                printf("duplicate values is %d\n", arr[i]);
                avail=1;
                break;
            }
        }
    }
    if(!avail){
        printf("No Duplicate Value Found");
    }
}
