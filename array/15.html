// Enter an array from user and print all the pallindrone numbers
#include 
int pallindrome(int num)
{
    int rev = 0, n = num;
    while (num != 0)
    {
        rev = (rev * 10) + (num % 10);
        num /= 10;
    }
    if (n == rev)
    {
        return 1;
    }
    return 0;
}
int main()
{
    int arr[5], i;
    for (i = 0; i < 5; i++)
    {
        printf("Enter the value: ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < 5; i++)
    {
        if (pallindrome(arr[i]))
            printf("%d is the pallindrome number in array\n", arr[i]);
    }
    return 0;
}
