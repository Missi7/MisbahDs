// Enter an array from user and print the maximum and minimum values
#include 
void main()
{
    int arr[5], i, n = 5, max, min;
    for (i = 0; i < 5; i++)
    {
        printf("Enter Value : ");
        scanf("%d", &arr[i]);
    }
    max = arr[0];
    for (i = 1; i < n; i++)
    {
        if (arr[i] > max)
        {
            max = arr[i];
        }
    }
    min = arr[0];
    for (i = 1; i < n; i++)
    {
        if (arr[i] < min)
        {
            min = arr[i];
        }
    }
    printf("The Minimum value in array is%d\n", min);
    printf("The Maximum value in array is %d", max);
}
