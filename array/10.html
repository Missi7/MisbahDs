// Enter an array from user and add all the values
#include 
void main()
{
    int arr[5], i, n = 5, sum = 0;
    for (i = 0; i < n; i++)
    {
        printf("Enter the value : ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < n; i++)
    {
        sum += arr[i];
    }
    printf("The sum of all the values is %d", sum);
}
