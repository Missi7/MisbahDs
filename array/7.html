// Enter an array from user and square all the values of array
#include 

void main()
{
    int arr[5], i, n = 5;
    for (i = 0; i < n; i++)
    {
        printf("Enter the value : ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < n; i++)
    {
        arr[i] = arr[i] * arr[i];
    }
    for (i = 0; i < n; i++)
    {
        printf("\nSquares of the array values are %d", arr[i]);
    }
}
