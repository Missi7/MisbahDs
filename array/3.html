// Enter an array from user and copy the array into second array
#include 

void main()
{
    int arr[5], n = 5;
    int arr1[n], i;
    for (i = 0; i < n; i++)
    {
        printf("Enter the value : ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < n; i++)
    {
        arr1[i] = arr[i];
    }

    printf("The values in first array");
    for (i = 0; i < n; i++)
    {
        printf("\n%d", arr[i]);
    }

    printf("\nThe values in second array");
    for (i = 0; i < n; i++)
    {
        printf("\n%d", arr1[i]);
    }
}
