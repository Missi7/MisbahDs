// Enter an array from user and print the even values
#include 

void main()
{
    int arr[5], i, n = 5;
    for (i = 0; i < n; i++)
    {
        printf("Enter the value : ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < n; i++)
    {
        if (arr[i] % 2 == 0)
        {
            printf("\nThe even values are %d", arr[i]);
        }
    }
}

Output:

Enter the value : 1
Enter the value : 2
Enter the value : 5
Enter the value : 3
Enter the value : 6

The even values are 2
The even values are 6
