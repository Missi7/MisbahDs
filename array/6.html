// Enter an array from user and print the positive values
#include 

void main()
{
    int arr[5], i, n = 5;
    for (i = 0; i < n; i++)
    {
        printf("Enter the value : ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < n; i++)
    {
        if (arr[i] > 0)
        {
            printf("\npositive value is %d", arr[i]);
        }
    }
}
