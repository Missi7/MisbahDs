// Enter two arrays from user and check they are same or not...
#include 
void main()
{
    int arr[5], arr1[5], n = 5, i;
    printf("Enter First Array:-\n");
    for (i = 0; i < n; i++)
    {
        printf("Enter the value :");
        scanf("%d", &arr[i]);
    }
    printf("Enter Second Array:-\n");
    for (i = 0; i < n; i++)
    {
        printf("Enter the value :");
        scanf("%d", &arr1[i]);
    }

    for (i = 0; i < n; i++)
    {
        if (arr[i] != arr1[i])
        {
            printf("Both the arrays are different :- ");
            break;
        }
    }

    if (i == n)
    {
        printf("they are same");
    }
}
