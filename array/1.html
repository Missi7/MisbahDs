// Enter an array from user and print all the values
#include 

void main()
{
    int arr[5], i;
    for (i = 0; i < 5; i++)
    {
        printf("Enter the value : ");
        scanf("%d", &arr[i]);
    }
    for (i = 0; i < 5; i++)
    {
        printf("\nThe Values are : %d", arr[i]);
    }
}
