// ENTER A MATRIX OF 4 ROWS AND 2 COLUMNS FROM USER AND DISPLAY IT IN MATRIX FROMAT
#include 
int main()
{
    int a[4][2], i, j;
    printf("enter the elements of matrix:\n");
    for (i = 0; i < 4; i++)
    {
        for (j = 0; j < 2; j++)
        {
            scanf("%d", &a[i][j]);
        }
    }
    printf("The formed matrix is....\n");
    for (i = 0; i < 4; i++)
    {
        for (j = 0; j < 2; j++)
        {
            printf("%d\t", a[i][j]);
        }
        printf("\n");
    }
}
